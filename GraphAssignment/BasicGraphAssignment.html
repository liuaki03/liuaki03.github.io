Q1: Which states are associated with the greatest number of United States presidents in terms of the presidents' birthplaces?¶

In [48]:
# I used a bar chart to show the frequency distribution of US presidential births in each state
# since the question asked which states had the greatest number of presidential births.

# Step 1: Create frequency df
presidents_freq = pd.value_counts(presidents_df["State"]).reset_index().rename(columns={"index":"State",
                                                                                        "State":"Count"})
# Step 2: Plot graph
fig,ax = plt.subplots()
ax.bar(presidents_freq["State"],presidents_freq["Count"])
fig.suptitle('Number of United States Presidents\' Births in Each State')
ax.xaxis.set_label_text('State') 
ax.yaxis.set_label_text('Number of Presidential Births') 
fig.set_size_inches(12,8)
plt.show()
